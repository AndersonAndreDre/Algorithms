#include <iostream>
#include <array>
using namespace std;

int recursiveLinear(int arr[],int n,int x){

    if(n==0){
    return -1;
}
else if(arr[n-1]==x){
    return n-1;
}
else{
    int res(recursiveLinear(arr,n-1,x));
}
}
  
  
int main(){
    int arr[7]={2,5,7,12,78,45,90};
    int x=78;
    int n=sizeof(arr);
    int res=recursiveLinear(arr,n,x);
    
    
    if(res==-1){
        cout<<"No such element"<<endl;
        
    }
    else{
        cout<<"element is at "<<res<<endl;
    }
    
    
}
